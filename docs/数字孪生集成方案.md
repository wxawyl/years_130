class TwinVisualParams {
  final TwinFaceState face;
  final TwinBodyState body;
  final TwinSkinState skin;

  final List&lt;TwinEffect&gt; effects;

  factory TwinVisualParams.fromMoodTheme(MoodTheme theme) {

      face: TwinFaceState.fromMoodType(theme.type),
      body: TwinBodyState.fromMoodType(theme.type),
      skin: TwinSkinState.fromMoodType(theme.type),
      moodType: theme.type,
      effects: _effectsFromMoodType(theme.type),
    );
  }
}

## 🎨 组件设计

### DigitalTwinWidget（核心组件）

```dart
class DigitalTwinWidget extends StatelessWidget {
  final double size;
  final VoidCallback? onTap;
  
  const DigitalTwinWidget({
    super.key,
    this.size = 180,
    this.onTap,
  });
  
  @override
  Widget build(BuildContext context) {
    return Consumer&lt;ThemeProvider&gt;(
      builder: (context, themeProvider, child) {
        final twinParams = TwinVisualParams.fromMoodTheme(
          themeProvider.currentTheme,
        );
        
        return GestureDetector(
          onTap: onTap,
          child: Container(
            width: size,
            height: size,
            child: Stack(
              children: [
                // 身体
                TwinBodyWidget(
                  params: twinParams.body,
                  size: size,
                ),
                // 面部
                TwinFaceWidget(
                  params: twinParams.face,
                  size: size,
                ),
                // 特效
                TwinEffectsWidget(
                  effects: twinParams.effects,
                  size: size,
                ),
              ],
            ),
          ),
        );
      },
    );
  }
}
```
